scripted_gui = {
    cgui_list_1 = {
    
        context_type = player_context
		
		window_name = "cgui_list"
		
		visible = {
            ROOT = {
				has_country_flag = open_culture_list
			}
		}
		
		# dynamic lists are used for populating grid boxes
		dynamic_lists = {
			# name of the grid box gui element
			cgui_list_grid = {
				array = array_culture #the array that will be used for populating the grid
				change_scope = no # since our array contains scope objects (countries) we can set change_scope to yes, which will chance current scope to array element while displaying the lists
				value = culture_gridbox_value
				entry_container = scripted_gui_example_7_country_entry # gui container name of the entry 
			}
		}
		
		effects = {
			CGUI_LIST_CLOSE_BUTTON_click = {
				ROOT = {
					clr_country_flag = open_culture_list
				}
			}
		}
		
		triggers = {
			culture_bozo_modz_list_visible = {
				check_variable = { culture_gridbox_value =  4 }
			}
			culture_greek_list_visible = {
				check_variable = { culture_gridbox_value = 1 }
			}
			culture_turkish_list_visible = {
				check_variable = { culture_gridbox_value = 2 }
			}
			culture_quebecois_list_visible = {
				check_variable = { culture_gridbox_value = 3 }
			}
		}
    }
}
